struct location{
    int npc_count;        // количество жителей
    int enemy_count;      // количество врагов
    std::string loc_name; // название локации
    int id;           // id локации
};

id должен быть у каждой локации, предмета, и npc.

для локаций используй однозначные (1, 0, 9)

для предметов четырёхзначные (0001, 0031, 1023)

для npc двухзначные (21, 42, 00)
